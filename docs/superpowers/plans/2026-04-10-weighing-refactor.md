# Weighing Refactor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Refactor the weighing module from N-assets-per-inbound interleaved model to 1-inbound = 1-asset with swap/direct/bulk weighing modes, add Container Registry, and create receipt-style weight bills.

**Architecture:** Each inbound represents a single container visit with exactly 2 weighings (gross + tare/swap). Three weighing modes (SWAP, DIRECT, BULK) determine how net weight is calculated. A new ContainerRegistry model enables dropdown selection for known containers. Weight bills are receipt-style PDFs with Dutch labels matching Pfister thermal printer output.

**Tech Stack:** PostgreSQL + Prisma ORM, Express.js, React 18, PDFKit, Vitest + Supertest

**Spec:** `docs/superpowers/specs/2026-04-10-weighing-refactor-design.md`

```javascript
const prisma = require('../utils/prismaClient');
const { writeAuditLog } = require('../utils/auditLog');

async function list(req, res, next) {
  try {
    const { search, active, page = 1, limit = 50 } = req.query;
    const pageNum = Math.max(1, parseInt(page, 10));
    const limitNum = Math.min(100, Math.max(1, parseInt(limit, 10)));
    const skip = (pageNum - 1) * limitNum;

    const where = {};
    if (search) {
      where.container_label = { contains: search, mode: 'insensitive' };
    }
    if (active !== undefined) {
      where.is_active = active === 'true';
    }

    const [data, total] = await Promise.all([
      prisma.containerRegistry.findMany({
        where,
        skip,
        take: limitNum,
        orderBy: { container_label: 'asc' },
      }),
      prisma.containerRegistry.count({ where }),
    ]);

    return res.json({ data, total, page: pageNum, limit: limitNum });
  } catch (err) {
    next(err);
  }
}

async function getById(req, res, next) {
  try {
    const container = await prisma.containerRegistry.findUnique({
      where: { id: req.params.id },
    });
    if (!container) return res.status(404).json({ error: 'Container not found' });
    return res.json(container);
  } catch (err) {
    next(err);
  }
}

async function create(req, res, next) {
  try {
    const { container_label, container_type, tare_weight_kg, volume_m3, notes } = req.body;
    if (!container_label || !container_type || tare_weight_kg == null) {
      return res.status(400).json({ error: 'container_label, container_type, and tare_weight_kg are required' });
    }

    const container = await prisma.$transaction(async (tx) => {
      const created = await tx.containerRegistry.create({
        data: { container_label, container_type, tare_weight_kg, volume_m3: volume_m3 || null, notes: notes || null },
      });
      await writeAuditLog({
        userId: req.user.userId,
        action: 'CREATE',
        entityType: 'ContainerRegistry',
        entityId: created.id,
        after: created,
      }, tx);
      return created;
    });

    return res.status(201).json(container);
  } catch (err) {
    if (err.code === 'P2002') return res.status(409).json({ error: 'Container label already exists' });
    next(err);
  }
}

async function update(req, res, next) {
  try {
    const { id } = req.params;
    const existing = await prisma.containerRegistry.findUnique({ where: { id } });
    if (!existing) return res.status(404).json({ error: 'Container not found' });

    const { container_label, container_type, tare_weight_kg, volume_m3, notes } = req.body;

    const container = await prisma.$transaction(async (tx) => {
      const updated = await tx.containerRegistry.update({
        where: { id },
        data: {
          ...(container_label !== undefined && { container_label }),
          ...(container_type !== undefined && { container_type }),
          ...(tare_weight_kg !== undefined && { tare_weight_kg }),
          ...(volume_m3 !== undefined && { volume_m3 }),
          ...(notes !== undefined && { notes }),
        },
      });
      await writeAuditLog({
        userId: req.user.userId,
        action: 'UPDATE',
        entityType: 'ContainerRegistry',
        entityId: id,
        before: existing,
        after: updated,
      }, tx);
      return updated;
    });

    return res.json(container);
  } catch (err) {
    if (err.code === 'P2002') return res.status(409).json({ error: 'Container label already exists' });
    next(err);
  }
}

async function remove(req, res, next) {
  try {
    const { id } = req.params;
    const existing = await prisma.containerRegistry.findUnique({ where: { id } });
    if (!existing) return res.status(404).json({ error: 'Container not found' });

    const container = await prisma.$transaction(async (tx) => {
      const updated = await tx.containerRegistry.update({
        where: { id },
        data: { is_active: false },
      });
      await writeAuditLog({
        userId: req.user.userId,
        action: 'DELETE',
        entityType: 'ContainerRegistry',
        entityId: id,
        before: existing,
        after: updated,
      }, tx);
      return updated;
    });

    return res.json(container);
  } catch (err) {
    next(err);
  }
}

module.exports = { list, getById, create, update, remove };
```

```javascript
async function recalculateInboundWeights(tx, inboundId) {
  const inbound = await tx.inbound.findUnique({
    where: { id: inboundId },
    include: {
      weighings: {
        include: { pfister_ticket: true },
        orderBy: { sequence: 'asc' },
      },
      assets: { orderBy: { sequence: 'asc' } },
    },
  });

  if (!inbound) return null;

  const mode = inbound.weighing_mode;
  const w1 = inbound.weighings.find((w) => w.sequence === 1);
  const w2 = inbound.weighings.find((w) => w.sequence === 2);

  for (const asset of inbound.assets) {
    let assetUpdate;

    if (mode && w1 && w2) {
      // New 1:1 mode (SWAP / DIRECT / BULK)
      const diff = roundWeight(Number(w1.weight_kg) - Number(w2.weight_kg));

      if (mode === 'DIRECT' && asset.estimated_tare_weight_kg) {
        assetUpdate = {
          gross_weighing_id: w1.id,
          tare_weighing_id: w2.id,
          gross_weight_kg: diff,
          tare_weight_kg: Number(asset.estimated_tare_weight_kg),
          net_weight_kg: roundWeight(diff - Number(asset.estimated_tare_weight_kg)),
        };
      } else {
        assetUpdate = {
          gross_weighing_id: w1.id,
          tare_weighing_id: w2.id,
          gross_weight_kg: Number(w1.weight_kg),
          tare_weight_kg: Number(w2.weight_kg),
          net_weight_kg: diff,
        };
      }
    } else {
      // Legacy interleaved mode (no weighing_mode set)
      const grossWeighing = inbound.weighings.find((w) => w.sequence === asset.sequence);
      const tareWeighing = inbound.weighings.find((w) => w.sequence === (asset.sequence || 0) + 1);
      assetUpdate = {
        gross_weighing_id: grossWeighing?.id || null,
        gross_weight_kg: grossWeighing ? Number(grossWeighing.weight_kg) : null,
        tare_weighing_id: tareWeighing?.id || null,
        tare_weight_kg: tareWeighing ? Number(tareWeighing.weight_kg) : null,
        net_weight_kg: grossWeighing && tareWeighing
          ? roundWeight(Number(grossWeighing.weight_kg) - Number(tareWeighing.weight_kg))
          : null,
      };
    }

    await tx.asset.update({ where: { id: asset.id }, data: assetUpdate });
  }

  // Inbound totals
  const firstWeighing = inbound.weighings[0];
  const tareWeighing = inbound.weighings.find((w) => w.is_tare) || null;
  const asset = inbound.assets[0];

  const inboundUpdate = {
    gross_weight_kg: firstWeighing ? Number(firstWeighing.weight_kg) : null,
    tare_weight_kg: tareWeighing ? Number(tareWeighing.weight_kg) : null,
    net_weight_kg: (mode && asset) ? asset.net_weight_kg : (
      firstWeighing && tareWeighing
        ? roundWeight(Number(firstWeighing.weight_kg) - Number(tareWeighing.weight_kg))
        : null
    ),
  };

  await tx.inbound.update({ where: { id: inboundId }, data: inboundUpdate });

  return tx.inbound.findUnique({
    where: { id: inboundId },
    include: INBOUND_INCLUDE,
  });
}
```

- [ ] **Step 1: Rewrite ticketGenerator.js**

Replace the entire `generateWeightTicket` function and helpers with a receipt-style PDF generator. Use PDFKit with narrower page width to mimic a receipt.

Key structure:

```javascript
const PDFDocument = require('pdfkit');
const prisma = require('../utils/prismaClient');

async function generateWeightTicket(inboundId) {
  const inbound = await prisma.inbound.findUnique({
    where: { id: inboundId },
    include: {
      order: { include: { supplier: true, carrier: true } },
      vehicle: true,
      assets: { include: { waste_stream: true, material_category: true } },
      weighings: { include: { pfister_ticket: true }, orderBy: { sequence: 'asc' } },
      confirmed_by_user: true,
    },
  });

  if (!inbound) throw new Error('Inbound not found');

  const asset = inbound.assets[0];
  const w1 = inbound.weighings.find((w) => w.sequence === 1);
  const w2 = inbound.weighings.find((w) => w.sequence === 2);

  // Receipt-style: 80mm wide ≈ 226pt, variable height
  const doc = new PDFDocument({
    size: [226, 600],
    margins: { top: 20, bottom: 20, left: 15, right: 15 },
  });

  const chunks = [];
  doc.on('data', (chunk) => chunks.push(chunk));

  // Header: Logo + company info
  drawReceiptHeader(doc);

  // Order details block
  drawReceiptOrderDetails(doc, inbound);

  // Separator
  doc.moveTo(15, doc.y).lineTo(211, doc.y).stroke();
  doc.moveDown(0.5);

  // Weighing 1
  if (w1) drawReceiptWeighing(doc, w1, '1. Gewicht');

  // Weighing 2
  if (w2) drawReceiptWeighing(doc, w2, '2. Gewicht');

  // Net weight section
  drawReceiptNetWeight(doc, inbound, asset);

  // Separator
  doc.moveTo(15, doc.y).lineTo(211, doc.y).stroke();
  doc.moveDown(0.5);

  // Footer
  drawReceiptFooter(doc, inbound);

  doc.end();

  return Buffer.concat(chunks);
}
```

- `drawReceiptHeader(doc)` — Statice logo, company name, address, phone
- `drawReceiptOrderDetails(doc, inbound)` — Leverancier, Referentie, Order, Inbound, Kenteken, Asset
- `drawReceiptWeighing(doc, weighing, label)` — Date, time, Volgnummer (ticket_number), Weg (device_id), weight
- `drawReceiptNetWeight(doc, inbound, asset)` — Netto Gewicht, plus DIRECT mode extras (Bruto Container, Container Tarra, Netto Lading)
- `drawReceiptFooter(doc, inbound)` — Bevestigd door, Gegenereerd timestamp
